CREATE TABLE titanic (
    passenger_id INT NOT NULL,
	survied  VARCHAR(50),
	class_of_travel VARCHAR(50),
    passenger_firstname VARCHAR(50),
    passenger_lastname VARCHAR(50),
	sex VARCHAR(20),
	age VARCHAR(50),
	number_of_sibspouse_aboard VARCHAR(50),
	number_of_parent_aboard VARCHAR(50),
	ticket VARCHAR(50),
	Fare VARCHAR(50),
	Cabin VARCHAR(50),
	Embarked VARCHAR(1),
	PRIMARY KEY (passenger_id)
)  ENGINE=INNODB;
